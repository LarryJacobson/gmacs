119  double pmin = 0.001;
120 
121 
122  jmin.allocate(r1,r2);

124 
125  for(int i = r1; i <= r2; i++ )
126  {
127 
128  dvector o = m_O(i);
129 
130  jmin(i) = c1+1; // index for min column
131  jmax(i) = c2; // index for max column
132  dvector cumsum = o/sum(o);
133  for(int j = c1+1; j <= c2; j++ )
134  {
135  cumsum(j) += cumsum(j-1);
136  cumsum(j) <= pmin ? jmin(i)++ : NULL;
137  j != c2 ? 1.0 - cumsum(j) < pmin ? jmax(i)-- : NULL : NULL;
138  }
139 
140  }
141 
142  // now allocate ragged arrays
143  m_Or.allocate(r1,r2,jmin,jmax);
144  m_Pr.allocate(r1,r2,jmin,jmax);
145  m_Or.initialize();
146  m_Pr.initialize();
147 
148  // fill ragged arrays
149  for(int i = r1; i <= r2; i++ )
150  {
151  dvector cso = m_O(i)/sum(m_O(i));
152  dvar_vector csp = m_P(i)/sum(m_P(i));
153  m_Or(i)(jmin(i),jmax(i)) = cso(jmin(i),jmax(i));
154  m_Pr(i)(jmin(i),jmax(i)) = csp(jmin(i),jmax(i));
155 
156  // add cumulative sum to tails.
157  m_Or(i)(jmin(i)) = sum(cso(c1,jmin(i)));
158  m_Or(i)(jmax(i)) = sum(cso(jmax(i),c2));
159  m_Pr(i)(jmin(i)) = sum(csp(c1,jmin(i)));
160  m_Pr(i)(jmax(i)) = sum(csp(jmax(i),c2));
